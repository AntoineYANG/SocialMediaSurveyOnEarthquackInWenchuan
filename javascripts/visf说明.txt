visf.ts 简单入门

引入声明
    /// <reference path='${local path}/visf.ts' />

创建二维坐标系
    方法
        constructor Visf.Axis.Axis2d(parent: JQuery<HTMLElement>, theme?: Visf.Color.Color): Visf.Axis.Axis2d
            parent: JQuery 选择集<svg>元素
            theme (可选): Visf.Color.Color 指定一个配色主题
    实例
        实例化一个没有主题的坐标系：
            let axis: Visf.Axis.Axis2d = new Visf.Axis.Axis2d($('#svg'));
        实例化一个主题为 Monet_bright 的坐标系：
            let axis: Visf.Axis.Axis2d = new Visf.Axis.Axis2d($('#poly_svg'), new Visf.Color.Artists.Monet.Monet_bright());
        
设置坐标系参数
    方法 (支持链式语法)
        Visf.Axis.Axis2d.set(param: string, value: string): Visf.Axis.Axis2d
    实例
        axis.set('margin', '0').set('margin', '40 60 40').set('fill', 'white');

获取坐标轴线段
    方法
        Visf.Axis.Axis2d.scale(limit?: string): JQuery<HTMLElement>
    实例
        let x_scale = axis.scale('x');

更改坐标轴
    方法 (支持链式语法)
        Visf.Axis.Axis2d.xScale(style: string): Visf.Axis.Axis2d
        Visf.Axis.Axis2d.yScale(style: string): Visf.Axis.Axis2d
    实例
        更改x轴为序数坐标轴，并指定定义域：
            axis.xScale('ordinal').among_x([2009, 2010, 2011, 2012, 2013, 2014, 2015, 2016, 2017, 2018, 2019]);

设置或更改定义域
    方法 (支持链式语法)
        Visf.Axis.Axis2d.domain_x(min: number, max: number): Visf.Axis.Axis2d
        Visf.Axis.Axis2d.domain_y(min: number, max: number): Visf.Axis.Axis2d
    实例
        设置y轴定义域：
            axis.domain_y(0, 1000);

更改坐标轴刻度显示
    方法
        Visf.Axis.Axis2d.note(ticks: number, limit?: string): JQuery<HTMLElement>
            ticks: number 显示刻度数量
            limit: 'x', 'y', 'both'<=default 作用坐标轴
    实例
        设置y轴显示的刻度数量，并修改样式：
            axis.note(8, 'y').css('fill', 'blue');

设置超限数值的处理原则
    方法 (支持链式语法)
        Visf.Axis.Axis2d.handle(solution: string, limit?: string): Axis2d
            solution: string
                'forbidden' - (原始值) 禁止输入超限数值，计算数值超限会抛出一个错误
                'hard' - 按照比例尺函数正常计算
                'stuck' - 作为定义域边界处理
    实例
        axis.handle('hard', 'y');

在坐标系中绘制图形
    方法
        Visf.Axis.Axis2d.append(element: string, ...data: number[]): JQuery<HTMLElement>
    实例
        axis.append('circle', [12, 10.8]).css('fill', 'yellow');

在坐标系中绘制路径
    方法
        Visf.Axis.Axis2d.path(data: number[][]): JQuery<HTMLElement>
    实例
        axis.path([[0, 0], [2, 10], [5, 20], [10, 50]]).css('stroke-width', '2');

在坐标系中绘制同一个系列的图形组
    方法
        Visf.Axis.Axis2d.join(element: string, nodes: number[][]): JQuery<HTMLElement>
    实例
        axis.join('circle', [[0, 0], [2, 10], [5, 20], [10, 50]]).attr('r', '10');

在坐标系中添加文字
    方法
        Visf.Axis.Axis2d.addtext(text: string, ...data: number[]): JQuery<HTMLElement>
    实例
        axis.addtext('value = 10.8', [12, 10.8]);

更改坐标系主题
    方法 (支持链式语法)
        Visf.Axis.Axis2d.fitIn(theme: Color.Color): Axis2d
    实例
        axis.fitIn(new Visf.Color.Artists.Matisse.Matisse_dark());
